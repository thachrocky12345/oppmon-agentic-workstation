# BM25 Search Implementation

## Overview

BM25 (Best Matching 25) is a probabilistic ranking function for text search. PostgreSQL implements this via `ts_rank` and `ts_rank_cd` functions with tsvector/tsquery.

## PostgreSQL Full-Text Search

### Basic Query

```sql
-- Simple search using plainto_tsquery
SELECT id, name,
       ts_rank_cd(search_vector, query, 32) as score
FROM skills,
     plainto_tsquery('english', 'git commit workflow') query
WHERE search_vector @@ query
  AND "tenantId" = 'tenant_123'
  AND "deletedAt" IS NULL
ORDER BY score DESC
LIMIT 20;
```

### Query Types

| Function | Use Case | Example |
|----------|----------|---------|
| `plainto_tsquery` | User input, auto-AND | `'git commit'` → `'git' & 'commit'` |
| `phraseto_tsquery` | Exact phrase | `'git commit'` → `'git' <-> 'commit'` |
| `websearch_to_tsquery` | Google-style | `'"exact" -exclude OR optional'` |
| `to_tsquery` | Manual control | `'git & (commit | push)'` |

### Normalization Options

The `ts_rank_cd` function accepts a normalization flag:

```sql
-- Normalization options (can be OR'd together):
-- 0: Default, no normalization
-- 1: Divides by 1 + log(document length)
-- 2: Divides by document length
-- 4: Divides by mean harmonic distance between extents
-- 8: Divides by number of unique words
-- 16: Divides by 1 + log(unique words)
-- 32: Divides by itself + 1 (recommended for most cases)

SELECT ts_rank_cd(search_vector, query, 32) as score
FROM skills, plainto_tsquery('english', 'git') query
WHERE search_vector @@ query;
```

## TypeScript Implementation

```typescript
// apps/api/src/lib/search/bm25.ts

import { prisma } from '@arkon/database';

export interface BM25Result {
  id: string;
  sourceType: string;
  score: number;
}

export interface BM25SearchOptions {
  tenantId: string;
  query: string;
  sourceTypes: string[];
  topK?: number;
  minScore?: number;
}

/**
 * Execute BM25 search across multiple source types
 */
export async function bm25Search(
  options: BM25SearchOptions
): Promise<BM25Result[]> {
  const { tenantId, query, sourceTypes, topK = 100, minScore = 0 } = options;

  // Build UNION query for all source types
  const queries: string[] = [];

  if (sourceTypes.includes('skill')) {
    queries.push(`
      SELECT id, 'skill' as "sourceType",
             ts_rank_cd(search_vector, query, 32) as score
      FROM skills,
           plainto_tsquery('english', $1) query
      WHERE search_vector @@ query
        AND "tenantId" = $2
        AND "deletedAt" IS NULL
    `);
  }

  if (sourceTypes.includes('mcp_server')) {
    queries.push(`
      SELECT id, 'mcp_server' as "sourceType",
             ts_rank_cd(search_vector, query, 32) as score
      FROM mcp_servers,
           plainto_tsquery('english', $1) query
      WHERE search_vector @@ query
        AND "tenantId" = $2
        AND "deletedAt" IS NULL
    `);
  }

  if (sourceTypes.includes('agent')) {
    queries.push(`
      SELECT id, 'agent' as "sourceType",
             ts_rank_cd(search_vector, query, 32) as score
      FROM agents,
           plainto_tsquery('english', $1) query
      WHERE search_vector @@ query
        AND "tenantId" = $2
    `);
  }

  if (sourceTypes.includes('workflow')) {
    queries.push(`
      SELECT id, 'workflow' as "sourceType",
             ts_rank_cd(search_vector, query, 32) as score
      FROM workflows,
           plainto_tsquery('english', $1) query
      WHERE search_vector @@ query
        AND "tenantId" = $2
    `);
  }

  if (queries.length === 0) {
    return [];
  }

  const unionQuery = `
    WITH all_results AS (
      ${queries.join(' UNION ALL ')}
    )
    SELECT id, "sourceType", score
    FROM all_results
    WHERE score >= $3
    ORDER BY score DESC
    LIMIT $4
  `;

  const results = await prisma.$queryRawUnsafe<BM25Result[]>(
    unionQuery,
    query,
    tenantId,
    minScore,
    topK
  );

  return results;
}

/**
 * Search a single source type (simpler API)
 */
export async function bm25SearchSingle(
  tenantId: string,
  sourceType: 'skill' | 'mcp_server' | 'agent' | 'workflow',
  query: string,
  topK: number = 100
): Promise<Array<{ id: string; score: number }>> {
  const tableName = {
    skill: 'skills',
    mcp_server: 'mcp_servers',
    agent: 'agents',
    workflow: 'workflows',
  }[sourceType];

  const deletedFilter = sourceType === 'skill' || sourceType === 'mcp_server'
    ? 'AND "deletedAt" IS NULL'
    : '';

  const results = await prisma.$queryRawUnsafe<Array<{ id: string; score: number }>>(
    `
    SELECT id, ts_rank_cd(search_vector, query, 32) as score
    FROM ${tableName},
         plainto_tsquery('english', $1) query
    WHERE search_vector @@ query
      AND "tenantId" = $2
      ${deletedFilter}
    ORDER BY score DESC
    LIMIT $3
    `,
    query,
    tenantId,
    topK
  );

  return results;
}
```

## Query Expansion Integration

Combine with taxonomy expansion for better recall:

```typescript
import { expandQuery } from './taxonomy.js';

export async function bm25SearchWithExpansion(
  tenantId: string,
  rawQuery: string,
  sourceTypes: string[],
  topK: number = 100
): Promise<BM25Result[]> {
  // Expand query terms
  const expandedTerms = expandQuery(rawQuery);
  const searchQuery = expandedTerms.join(' | '); // OR for expanded terms

  // Use websearch_to_tsquery for OR logic
  const results = await prisma.$queryRaw`
    SELECT id, 'skill' as "sourceType",
           ts_rank_cd(search_vector, query, 32) as score
    FROM skills,
         websearch_to_tsquery('english', ${searchQuery}) query
    WHERE search_vector @@ query
      AND "tenantId" = ${tenantId}
      AND "deletedAt" IS NULL
    ORDER BY score DESC
    LIMIT ${topK}
  `;

  return results;
}
```

## Weighting by Field

Use tsvector weights to prioritize matches:

```sql
-- A (name) matches weighted 1.0
-- B (description) matches weighted 0.4
-- C (content) matches weighted 0.2
-- D (metadata) matches weighted 0.1

-- Customize weights in ts_rank
SELECT id,
       ts_rank(
         '{0.1, 0.2, 0.4, 1.0}',  -- {D, C, B, A} weights
         search_vector,
         query
       ) as score
FROM skills,
     plainto_tsquery('english', 'git') query
WHERE search_vector @@ query;
```

## Performance Tips

### 1. Use GIN Index

```sql
CREATE INDEX idx_skills_search ON skills USING gin(search_vector);
```

### 2. Limit Result Set Early

```sql
-- Bad: ranks all matches
SELECT * FROM skills WHERE search_vector @@ query ORDER BY score DESC;

-- Good: limits before ranking
WITH matches AS (
  SELECT id, search_vector
  FROM skills
  WHERE search_vector @@ query
  LIMIT 1000  -- Cap matches
)
SELECT id, ts_rank_cd(search_vector, query, 32) as score
FROM matches
ORDER BY score DESC
LIMIT 20;
```

### 3. Pre-filter by Tenant

```sql
-- Bad: searches all, then filters
SELECT * FROM skills
WHERE search_vector @@ query AND "tenantId" = 'x';

-- Good: index on (tenantId, search_vector)
CREATE INDEX idx_skills_tenant_search
ON skills ("tenantId") INCLUDE (search_vector);
```

## Testing BM25

```typescript
// apps/api/src/lib/search/bm25.test.ts

import { describe, it, expect } from 'vitest';
import { bm25SearchSingle } from './bm25.js';

describe('bm25Search', () => {
  it('finds exact name matches', async () => {
    const results = await bm25SearchSingle(
      'tenant_test',
      'skill',
      'Git Commit Helper',
      10
    );

    expect(results.length).toBeGreaterThan(0);
    expect(results[0].score).toBeGreaterThan(0);
  });

  it('ranks exact matches higher than partial', async () => {
    const results = await bm25SearchSingle(
      'tenant_test',
      'skill',
      'git commit',
      10
    );

    // Exact match should be first
    // Partial matches should follow
    expect(results[0].score).toBeGreaterThan(results[1]?.score || 0);
  });

  it('returns empty for no matches', async () => {
    const results = await bm25SearchSingle(
      'tenant_test',
      'skill',
      'xyznonexistent123',
      10
    );

    expect(results).toEqual([]);
  });
});
```
